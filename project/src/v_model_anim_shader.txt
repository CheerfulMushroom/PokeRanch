#version 330 core

layout (location = 0) in vec3 position;
layout (location = 1) in vec3 normal;
layout (location = 2) in vec2 tex_coords;
layout (location = 3) in ivec4 bone_IDs;
layout (location = 4) in vec4 weights;

out vec2 out_tex_coords;

const int MAX_BONES = 100;


uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

uniform mat4 bones_array[MAX_BONES];

void main() {

    mat4 bone_transform = bones_array[bone_IDs[0]] * weights[0];
    bone_transform += bones_array[bone_IDs[1]] * weights[1];
    bone_transform += bones_array[bone_IDs[2]] * weights[2];
    bone_transform += bones_array[bone_IDs[3]] * weights[3];

    out_tex_coords = tex_coords;

    vec4 PosL = bone_transform * vec4(position, 1.0);

    gl_Position = projection * view * model * PosL;
}
